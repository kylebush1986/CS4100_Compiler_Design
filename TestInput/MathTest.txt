UNIT MATHTEST;

VAR TOTAL, SUM, PRODUCT, QUOTIENT, DIFFERENCE : INTEGER;

BEGIN
  SUM := 3 + 2;
  PRODUCT := 4 * 5;
  QUOTIENT := 15 / 3;
  DIFFERENCE := 12 - 7;
  TOTAL := (DIFFERENCE + QUOTIENT) * PRODUCT - SUM;
  WRITELN(SUM);
  WRITELN(PRODUCT);
  WRITELN(QUOTIENT);
  WRITELN(DIFFERENCE);
  WRITELN(TOTAL)
END.
